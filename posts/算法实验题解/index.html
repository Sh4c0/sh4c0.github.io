算法课实验一：勇者斗恶龙

传送门

题意

略

思路

贪心的想，我们需要让勇士的能力值浪费的尽可能小，所以我们应该选择一个能力值与巨龙防御力相近的勇士出门迎战。

简单的证明：

假设巨龙防御力为 $x$，攻击力为 $y$，比巨龙攻击力小的勇士中攻击力最大的为 $a_i$，而存在 $a_j<a_i$

如果我们选择 $a_i$ 出战，$a_i$ 需要喝的药为 $x-a_i$，如果我们选择 $a_j$ 出战，$a_j$ 需要喝的药为 $x-a_j$，这里 $a_i$ 比 $a_j$ 节约了 $a_i-a_j$ 瓶药

考虑防御的战士，选择 $a_i$ 出战防御的战士能力之和为 $sum-a_i$，选择 $a_j$ 出战防御的战士能力之和为 $sum-a_j$，显然 $sum-a_i<sum-a_j$

如果 $y<sum-a_i<sum-a_j$ ，那么无论选 $a_i$ 和 $a_j$ 出战，防御的战士能力值之和都能满足，$a_i$则由于出战时节约了 $a_i-a_j$ 瓶药优于 $a_j$

如果 $sum-a_i<y<sum-a_j$ ，那么这里选择 $a_i$ 出战需要再花费 $y-sum+a_i$ 瓶药，显然 $y-sum+a_i<a_i-a_j$，因此 $a_i$出战仍然优于 $a_j$

如果 $sum-a_i<sum-a_j<y$，那么这里选择 $a_i$ 出战需要比 $a_j$ 多花费 $(sum-a_j)-(sum-a_i)$ 瓶药，即 $a_i-a_j$，显然此时两者没有区别

因此整体而言，对于能力值比巨龙防御力小的勇士，我们应该选择能力值最大的那一个。

对于能力值比巨龙大的勇士，我们可以通过同样的方法证得应选择能力值最小的那一个。

而对于如何找到能力值最相近的那个勇士，我们可以通过二分法找到。

总时间复杂度 $O(mlogn)$

代码

#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
typedef long long ll;
const int maxn=2e6+5;

ll n,m;
ll a[maxn];

inline void solve(){
    cin>>n;
    ll sum=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        sum+=a[i];
    }
    sort(a+1,a+1+n);
    cin>>m;
    while(m--){
        ll x,y,l=1,r=n;
        cin>>x>>y;
        while(l<r){
            ll mid=(l+r)>>1;
            if(a[mid]>=x)r=mid;
            else l=mid+1;
        }
        if(l!=1)cout<<min(max(0ll,x-a[l])+max(0ll,y-(sum-a[l])),max(0ll,x-a[l-1])+max(0ll,y-(sum-a[l-1])))<<endl;
        else cout<<max(0ll,x-a[l])+max(0ll,y-(sum-a[l]))<<endl;
    }
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int __=1;
    while(__--)solve();
    return 0;
}

算法课实验二：挖隧道

传送门

题意

略

思路

由题意，定义 $dp[i]$ 为到 $i$ 米的最小花费

显然有如下转移方程 $$ dp[i]=Min(dp[i-1]+a[i],dp[i-1]+60,dp[i-3]+150) $$ 注意边界条件即可

代码

#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
typedef long long ll;
const int maxn=2e6+5;

ll n,m;
ll a[maxn],dp[maxn];

inline void solve(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        dp[i]=min(dp[i-1]+a[i],dp[i-1]+60);
        if(i>=3)dp[i]=min(dp[i],dp[i-3]+150);
    }
    cout<<dp[n]<<endl;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int __=1;
    while(__--)solve();
    return 0;
}

算法课实验三：消消乐

题意

略

思路

显然最小生成树

我们考虑Kruskal和prim，在边较多的稠密图或者完全图中，考虑边的kruskal复杂度显然是比考虑点的prim差

代码

#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
const int inf=0x3f3f3f3f;
const int maxn=5e3+5;

int n,m;
int g[maxn][maxn];
int dis[maxn];

void init(){}

inline void solve(){
    cin>>n>>m;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            cin>>g[i][j];
        }
        if(i==1)for(int j=1;j<=n;j++)dis[j]=g[i][j];
    }
    for(int i=2;i<=n;i++){
        int mnid,mndis=inf;
        for(int j=2;j<=n;j++){
            if(dis[j]==-1)continue;
            if(dis[j]<mndis){
                mndis=dis[j];
                mnid=j;
            }
        }
        m+=mndis;
        for(int j=1;j<=n;j++){
            if(dis[j]==-1)continue;
            dis[j]=min(dis[j],g[mnid][j]);
        }
    }
    cout<<m<<endl;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int __=1;
    while(__--)solve();
    return 0;
}

算法课实验四：ShacozzZ的生财之道

传送门

题意

略

思路

对于最下面的小球，显然他有两种掉落的方法（左边or右边）

考虑最下面的两个小球，显然有四种掉落的方法

不难看出掉落的种类一定是 $2^{10}$ 种，即分母为 $2^{10}$

而对于分子，我们只需要利用深度优先搜索枚举所有情况即可

那么我们枚举情况的时候，应该统计ShacozzZ赢的情况还是输的情况呢

显然两种都可以

如果统计ShacozzZ输的情况，那么我们需要走到答案树的最底层才能确定

而如果统计ShacozzZ的情况，我们只需要走到一个出现了降序的地方即可判断答案树的这一棵子树均为必输的情况，因此效率是优于上面的做法的

代码

#include<bits/stdc++.h>
using namespace std;

int a[15];
vector<int> l,r;
int up,down;

void dfs(int t){
    if(t==11){
        bool lose=1;
        for(int i=1;i<l.size();i++)if(l[i]<l[i-1])lose=0;
        for(int i=1;i<r.size();i++)if(r[i]<r[i-1])lose=0;
        if(lose)up--;
        return ;
    }
    l.push_back(a[t]);
    dfs(t+1);
    l.erase(l.end()-1);
    r.push_back(a[t]);
    dfs(t+1);
    r.erase(r.end()-1);
}

ll gcd(ll a,ll b) {
    return b>0?gcd(b,a%b):a;
}

inline void solve(){
    up=down=(1<<10);
    l.clear();
    r.clear();
    for(int i=1;i<=10;i++)cin>>a[i];
    reverse(a+1,a+11);
    dfs(1);
    cout<<up/gcd(up,down)<<" "<<down/gcd(up,down)<<endl;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int __=1;
    cin>>__;
    while(__--)solve();
    return 0;
}

#include<bits/stdc++.h>
using namespace std;

int a[15];
vector<int> l,r;
int up,down;

void dfs(int t){
    bool win=0;
    if(l.size()>=2&&l[l.size()-1]<l[l.size()-2])win=1;
    if(r.size()>=2&&r[r.size()-1]<r[r.size()-2])win=1;
    if(win){
        up+=(1<<(10-t+1));
        return ;
    }
    if(t==11)return ;
    l.push_back(a[t]);
    dfs(t+1);
    l.erase(l.end()-1);
    r.push_back(a[t]);
    dfs(t+1);
    r.erase(r.end()-1);
}

ll gcd(ll a,ll b) {
    return b>0?gcd(b,a%b):a;
}

inline void solve(){
    up=0;
    down=(1<<10);
    l.clear();
    r.clear();
    for(int i=1;i<=10;i++)cin>>a[i];
    reverse(a+1,a+11);
    dfs(1);
    cout<<up/gcd(up,down)<<" "<<down/gcd(up,down)<<endl;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int __=1;
    cin>>__;
    while(__--)solve();
    return 0;
}

算法课实验五：蜘蛛纸牌

传送门

题意

略

思路

暴力枚举

or

随机化（随机化就不会输）

代码

#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
const int inf=0x3f3f3f3f;
const int maxn=2e6+5;

int n,m;
int a[maxn];
vector<int> l,r;
int ans=inf;

bool check(){
    for(int i:l)if(i!=0)return 0;
    for(int i:r)if(i!=0)return 0;
    return 1;
}

void gao(int t){
    if(check()){
        ans=min(ans,t);
        return ;
    }
    if(t>=ans)return ;
    for(int i=0;i<l.size();i++){
        if(l[i]>0){
            for(int j=0;j<r.size();j++){
                if(r[j]<0){
                    int d=min(abs(l[i]),abs(r[j]));
                    l[i]-=d;
                    r[j]+=d;
                    gao(t+1);
                    l[i]+=d;
                    r[j]-=d;
                }
            }
        }
    }
}

inline void solve(){
    cin>>n;
    m=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        m+=a[i];
    }
    m/=n;
    for(int i=1;i<=n;i++){
        a[i]-=m;
        if(a[i]>0)l.push_back(a[i]);
        else if(a[i]<0)r.push_back(a[i]);
    }
    gao(0);
    cout<<ans<<endl;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    int __=1;
    while(__--)solve();
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
typedef long long ll;

int n,m;
int ans;
int a[15],b[15];

int f(){
    return rand()%n+1;
}

bool check(){
    for(int i=1;i<=n;i++)if(a[i]!=0)return 0;
    return 1;
}

void ff(){
    for(int T=0;T<ans;T++){
        if(check()){
            ans=T;
            return ;
        }
        int u,v;
        while(1){
            u=f();
            if(a[u]>0)break;
        }
        while(1){
            v=f();
            if(a[v]<0)break;
        }
        int d=min(abs(a[u]),abs(a[v]));
        a[u]-=d;
        a[v]+=d;
    }
}

inline void solve(){
    cin>>n;
    m=0;
    ans=n-1;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        b[i]=a[i];
        m+=a[i];
    }
    m/=n;
    for(int i=1;i<=n;i++){
        a[i]-=m;
        b[i]-=m;
    }
    int run=30;
    while(run--){
        for(int i=1;i<=n;i++)a[i]=b[i];
        ff();
    }
    cout<<ans<<endl;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    srand(time(0));
    int __=1;
    while(__--)solve();
    return 0;
}
